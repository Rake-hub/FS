private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        
        Coche c=null;
        String matricula=jTextField4.getText();
        String marca=jTextField1.getText();
        String modelo=jTextField2.getText();
        int motor=Integer.parseInt( jTextField3.getText());
        
        
        boolean existe=false;
        
        c=new Coche(matricula,marca,modelo,motor);
        
        if (coches.size() != 0) {

            for (int i = 0; i < coches.size(); i++) {
                if (coches.get(i).getMatricula().equals(matricula)) {
                    existe = true;
                }
            }//for
        if (!existe) {
            coches.add(c);
            JOptionPane.showMessageDialog(this, "Coche creado correctamente");
            JOptionPane.showMessageDialog(this, "Número de coches creados " + coches.size());
            jTextField1.setText("");
            jTextField2.setText("");
            jTextField3.setText("");
            jTextField4.setText("");

        } else {

            JOptionPane.showMessageDialog(this, "Matrícula ya existe");
             jTextField4.setText("");

        }
        } else {

            coches.add(c);
            JOptionPane.showMessageDialog(this, "Coche creado correctamente");
            JOptionPane.showMessageDialog(this, "Número de coches creados " + coches.size());
            jTextField1.setText("");
            jTextField2.setText("");
            jTextField3.setText("");
            jTextField4.setText("");
        }

    }                                        

    private void formWindowOpened(java.awt.event.WindowEvent evt) {                                  
        // TODO add your handling code here:
        coches = new ArrayList();
    }                                 

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:

        FileOutputStream fos;
        ObjectOutputStream oos;

        try {
            fos = new FileOutputStream("coches.dat");
            oos = new ObjectOutputStream(fos);
            oos.writeObject(coches);
            JOptionPane.showMessageDialog(this, "ArrayList guardado");

        } catch (IOException ioe) {
            System.out.println("No se pudo crear el fichero" + ioe.getMessage());
        }


    }
